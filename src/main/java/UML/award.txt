state 订单结束 {
  state 发放成功 #pink{
  }
  发放成功: 奖励正常发放
  state 无效订单 #pink{
  }
  无效订单: 风控等原因产生的无效订单
}

state 订单执行中 {
  state 订单未执行 #green {
  }
  订单未执行: 订单刚创建
  state 发放失败 #green{
  }
  发放失败: 奖励发放失败
}

[*] -[#blue]-> 订单未执行: 入口
订单未执行 --> 发放成功
订单未执行 --> 无效订单

订单未执行 --> 发放失败
发放失败 --> 发放失败: 重试任务
发放失败 --> 发放成功: 重试任务
发放失败 --> 无效订单: 重试任务

@enduml
